<?php
// On prpare l'adresse  rechercher
$address = "Palais de l'Elyse 55, rue du faubourg Saint-Honor 75008 Paris";
 
// On prpare l'URL du gocodeur
$geocoder = "http://maps.googleapis.com/maps/api/geocode/json?address=%s&sensor=false";
 
// Pour cette exemple, je vais considrer que ma chane n'est pas
// en UTF8, le gocoder ne fonctionnant qu'avec du texte en UTF8
$url_address = utf8_encode($address);
 
// Penser a encoder votre adresse
$url_address = urlencode($url_address);
 
// On prpare notre requte
$query = sprintf($geocoder,$url_address);
 
// On interroge le serveur
$results = file_get_contents($query);
 
// On affiche le rsultat
var_dump(json_decode($results));

?>
